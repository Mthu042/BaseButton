import { Canvas, Meta, Story, ArgsTable } from "@storybook/addon-docs";

import { useState, useCallback } from "react";
import { Button } from "./Button";
import { AddIcon, RightArrowIcon, ProfileIcon } from "rupor-icons";

<Meta title="Atoms/Buttons/Button" component={Button} />

export const Template = (args) => <Button {...args} />;

export const LoadingButton = () => {
  const [loading, setLoading] = useState(false);
  return (
    <div>
      <Button
        theme="primary"
        title="start loading"
        isLoading={loading}
        onClick={() => setLoading(!loading)}
        size={"medium"}
        loaderIcon="loading..."
      />
    </div>
  );
};

# Button

Кнопка имеет несколько тем и состояний описанных ниже, так же конпке всегда можно задать свойство `backgroundColor`.

## Тема

Передав свойство `theme` можно поменять стили кнопки, на данный момент доступно
два варинта темы и изменения затрагивают цвет и ховер.

- **Primary Medium**

Передав свойство `size` можно изменить размер конпки, есть 3 размера `medium (40px), large (52px), и значение по умолчанию если не одно из них не заданно (48px)`.
Передав свойство `isLoading` копка будет показывать иконку переданую в `loaderIcon` в зависимости от значения `true` или `false`.
